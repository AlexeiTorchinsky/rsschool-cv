              //Implementing Object.create
              Object.create = function(prototype, properties) {
                if (typeof prototype !== 'object' ||  prototype === undefined ){
                  throw new TypeError();
                }
                let obj = new Object;
                obj.__proto__ = prototype;
                if(properties !== undefined) {
                  Object.defineProperties(obj, properties);
                }
                return obj;
              };
